  @Override
  public void OnRegisterMsg(int ppl_code, int obj_type, int face_id, int time_out, byte[] ft, byte[] rgb_img, byte[] ir_img) {
      if(mIsRegister == false) return;

      if(ppl_code == 96) {  // 96 = 注册成功
          // ← ft 就是从设备获取的特征值
          int ret = mUserManager.addUser(face_id, obj_type,
              String.format("%d_%s", face_id, obj_type == 0 ? "face" : "palm"), ft);

          if(ret >= 0) {
              // 保存到本地文件
              mUserManager.saveBook("/storage/emulated/0/Documents/Users.dat");
              // 显示"录入成功"
          }
      }
  }

  3. 特征值存储 - UserManager.java:103-111

  public int addUser(int face_id, int type, String name, byte[] feature) {
      w.lock();
      // ← 创建用户，保存特征值
      User user = new User(face_id, name, type, feature);
      UserMap.put(face_id, user);
      user_list[face_id] = 1;
      w.unlock();
      return 1;
  }

  根据代码分析，人脸数据同步需要使用以下接口：

  A 机器（导出人脸特征）

  代码位置: facepass/src/main/java/com/example/facepass/NativeLib.java

  在 A 机器录入人脸后，通过以下方式获取特征数据：

  // 1. 录入人脸
  nativeLib.StartRegister(face_id, obj_type, time_out);

  // 2. 在回调中获取特征数据
  nativeLib.setFacePassCallBack(new NativeLib.facePassCallBack() {
      @Override
      public void OnRegisterMsg(int ppl_code, int obj_type, int face_id,
                                int time_out, byte[] ft, byte[] rgb_img, byte[] ir_img)

          // ft 就是人脸特征数据，需要保存下来用于同步
          // face_id 是用户ID
          // obj_type 是对象类型 (0=人脸, 1=掌纹)
      }
  });

  或者通过 AddUserByJpeg 直接从图片获取特征：

  int ret = nativeLib.AddUserByJpeg(face_id, jpegImage);
  if (ret >= 0) {
      // 特征数据在 nativeLib.ft 字段中
      byte[] feature = nativeLib.ft;  // 保存这个特征数据
      int objType = nativeLib.obj_type;
  }
